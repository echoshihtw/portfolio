% Tight résumé layout — reduces Pandoc's loose defaults.
% Narrow side padding (was ~1in), compact lists + section spacing.

\usepackage[left=0.5in,right=0.5in,top=0.45in,bottom=0.4in]{geometry}

\usepackage{enumitem}
\setlist[itemize]{leftmargin=1.1em, itemsep=1pt, topsep=1.5pt, parsep=0pt}

\usepackage{titlesec}
\titleformat{\section}{\large\bfseries}{}{0pt}{}[\vspace{-4pt}\titlerule]
\titleformat{\subsection}{\normalsize\bfseries}{}{0pt}{}
\titlespacing*{\section}{0pt}{6pt}{2pt}
\titlespacing*{\subsection}{0pt}{4pt}{1pt}

\setlength{\parskip}{2pt}
\setlength{\parindent}{0pt}

% ATS-friendly: disable hyphenation + ragged-right so no word is split
% across lines (keeps résumé-scanner text extraction clean).
\usepackage[none]{hyphenat}
\usepackage{ragged2e}
\RaggedRight
